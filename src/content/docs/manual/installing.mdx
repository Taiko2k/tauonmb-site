import { Tabs, TabItem, Card, LinkButton } from '@astrojs/starlight/components';


<Tabs>
  <TabItem label="Arch Linux" icon="linux">
    <Card title="Install on Arch Linux">
      Install on Arch based distros using Pacman
      ```bash
      sudo pacman -Syu tauon-music-box
      ```
    </Card>
  </TabItem>
  <TabItem label="Flatpak" icon="linux">
    <Card title="Install via Flatpak">
      Install on most Linux distros using Flatpak

      ```bash
      # Install
      flatpak install com.github.taiko2k.tauonmb

      # Give access to external music library
      flatpak override com.github.taiko2k.tauonmb --filesystem=/mnt/example

      # Uninstall
      flatpak uninstall com.github.taiko2k.tauonmb
      ```

    </Card>
  </TabItem>
  <TabItem label="Windows" icon="seti:windows">
    <Card title="Install on Windows">
      Download an installer from the releases page:
      <LinkButton href="https://github.com/Taiko2k/TauonMusicBox/releases" variant="minimal">Github Releases</LinkButton>
    </Card>
  </TabItem>
  <TabItem label="macOS" icon="apple">
    <Card title="Install on macOS">
      Download latest DMG: [Tauon v9.1.2 DMG](https://github.com/Taiko2k/Tauon/releases/download/v9.1.2/Tauon-v9.1.2-signed.dmg)

      Or download an older release from the [releases page](https://github.com/Taiko2k/TauonMusicBox/releases).
    </Card>
  </TabItem>
</Tabs>
